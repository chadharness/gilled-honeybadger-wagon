### Example 2: CTE with growth calculation
Query: "List the 5 client segments with highest deposit growth quarter-to-date"
```sql
WITH current_period AS (
    SELECT
        LINE_OF_BUSINESS_LEVEL_2_NAME,
        SUM(NET_AMOUNT) AS current_total
    FROM deposits_data_expanded
    WHERE SETTLEMENT_DATE BETWEEN DATE '2024-07-01' AND DATE '2024-09-30'
    GROUP BY LINE_OF_BUSINESS_LEVEL_2_NAME
),
prior_period AS (
    SELECT
        LINE_OF_BUSINESS_LEVEL_2_NAME,
        SUM(NET_AMOUNT) AS prior_total
    FROM deposits_data_expanded
    WHERE SETTLEMENT_DATE BETWEEN DATE '2024-04-01' AND DATE '2024-06-30'
    GROUP BY LINE_OF_BUSINESS_LEVEL_2_NAME
)
SELECT
    c.LINE_OF_BUSINESS_LEVEL_2_NAME,
    ROUND(c.current_total / 1000000, 2) AS current_mm,
    ROUND(p.prior_total / 1000000, 2) AS prior_mm,
    ROUND((c.current_total - COALESCE(p.prior_total, 0)) / 1000000, 2) AS growth_mm
FROM current_period c
LEFT JOIN prior_period p ON c.LINE_OF_BUSINESS_LEVEL_2_NAME = p.LINE_OF_BUSINESS_LEVEL_2_NAME
ORDER BY ABS(c.current_total - COALESCE(p.prior_total, 0)) DESC
LIMIT 5
```

### Example 4: Variance analysis with temporal grouping
Query: "Which LOBs had the largest variance in deposits month-over-month?"
```sql
WITH monthly AS (
    SELECT
        EXTRACT(YEAR FROM SETTLEMENT_DATE) AS yr,
        EXTRACT(MONTH FROM SETTLEMENT_DATE) AS mo,
        LINE_OF_BUSINESS_LEVEL_2_NAME AS lob,
        SUM(NET_AMOUNT) AS monthly_net
    FROM deposits_data_expanded
    WHERE SETTLEMENT_DATE BETWEEN DATE '2024-01-01' AND DATE '2024-09-30'
    GROUP BY
        EXTRACT(YEAR FROM SETTLEMENT_DATE),
        EXTRACT(MONTH FROM SETTLEMENT_DATE),
        LINE_OF_BUSINESS_LEVEL_2_NAME
)
SELECT
    lob,
    COUNT(*) AS months_with_data,
    ROUND(AVG(monthly_net) / 1000000, 2) AS avg_monthly_net_mm,
    ROUND(STDDEV_POP(monthly_net) / 1000000, 2) AS stddev_monthly_net_mm
FROM monthly
GROUP BY lob
HAVING COUNT(*) >= 3
ORDER BY STDDEV_POP(monthly_net) DESC
```
